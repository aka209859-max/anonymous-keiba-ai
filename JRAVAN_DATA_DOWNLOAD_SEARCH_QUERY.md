#### 4.3 サンプルコード（15年分データ取得）
```python
# 期待されるサンプルコード（具体的なコード例が欲しい）
import jvlink  # または該当するSDK
from datetime import datetime, timedelta
import pandas as pd

# 認証
api_key = "YOUR_API_KEY"
jv = jvlink.initialize(api_key)

# データ取得（2010年1月1日〜2025年12月31日、15年分）
start_date = "2010-01-01"
end_date = "2025-12-31"

venues = ["東京", "中山", "阪神", "京都", "中京", "新潟", "福島", "札幌", "函館", "小倉"]

print(f"データ取得開始: {start_date} 〜 {end_date}")

# 全データ取得
data = jv.get_race_data(
    start_date=start_date,
    end_date=end_date,
    venues=venues,
    include_odds=True,  # オッズ情報を含める
    include_pre_day_odds=True,  # 前日オッズを含める
    include_payouts=True,  # 払戻金情報を含める
    include_training=True,  # 調教情報を含める
    include_all=True  # 全データを取得
)

print(f"取得完了: {len(data)}件")

# CSV保存
data.to_csv("data/jravan/raw/jra_2010_2025_all.csv", encoding='shift-jis')

# 年別に分割保存（推奨）
for year in range(2010, 2026):
    df_year = data[data['kaisai_nen'] == year]
    output_path = f"data/jravan/raw/{year}/jra_{year}_all.csv"
    df_year.to_csv(output_path, encoding='shift-jis')
    print(f"保存完了: {output_path} ({len(df_year)}件)")
```
